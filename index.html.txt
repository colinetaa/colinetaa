<script>

let balance = localStorage.getItem("minerBalance");

if(balance === null){
balance = 1000;
}else{
balance = Number(balance);
}

let bet = 0;
let mines = [];
let mineCount = 3;
let active = false;
let opened = 0;

const grid = document.getElementById("grid");

function updateBalance(){

if(balance <= 0){
balance = 100;
alert("Баланс пополнен до $100");
}

document.getElementById("money").innerText = balance;

localStorage.setItem("minerBalance", balance);

}

function createGrid(){

grid.innerHTML = "";

for(let i=0;i<25;i++){

let cell = document.createElement("button");

cell.className="cell";

cell.onclick = function(){
openCell(i,cell);
}

grid.appendChild(cell);

}

}

function startGame(){

bet = Number(document.getElementById("bet").value);

mineCount = Number(document.getElementById("mines").value);

if(!bet || bet <= 0){
alert("Введите ставку");
return;
}

if(bet > balance){
alert("Недостаточно денег");
return;
}

balance -= bet;

updateBalance();

opened = 0;

document.getElementById("profit").innerText = bet;

mines = [];

while(mines.length < mineCount){

let r = Math.floor(Math.random()*25);

if(!mines.includes(r)){
mines.push(r);
}

}

active = true;

createGrid();

}

function openCell(i,cell){

if(!active) return;

if(cell.innerText !== "") return;

if(mines.includes(i)){

cell.innerText="💣";
cell.style.background="red";

alert("Вы проиграли");

active=false;

return;

}

cell.innerText="💎";
cell.style.background="green";

opened++;

let win = bet * (1 + opened * 0.35);

document.getElementById("profit").innerText = win.toFixed(2);

}

function cashout(){

if(!active) return;

let win = Number(document.getElementById("profit").innerText);

balance += win;

updateBalance();

alert("Вы забрали $"+win);

active=false;

}

createGrid();

updateBalance();

</script>
