<script>

let balance=localStorage.getItem("balance")

if(!balance){
balance=1000
}else{
balance=Number(balance)
}

let bet=0
let mines=[]
let opened=0
let mineCount=3
let active=false

function save(){
localStorage.setItem("balance",balance)
}

function updateBalance(){

if(balance<=0){
balance=100
alert("Баланс пополнен до 100$")
}

document.getElementById("money").innerText=balance
save()

}

function createGrid(){

let grid=document.getElementById("grid")
grid.innerHTML=""

for(let i=0;i<25;i++){

let btn=document.createElement("button")

btn.className="cell"

btn.onclick=function(){
openCell(i,btn)
}

grid.appendChild(btn)

}

}

function startGame(){

bet=Number(document.getElementById("bet").value)
mineCount=Number(document.getElementById("mines").value)

if(bet<=0){
alert("Введите ставку")
return
}

if(bet>balance){
alert("Недостаточно денег")
return
}

balance-=bet
updateBalance()

opened=0
active=true
mines=[]

while(mines.length<mineCount){

let r=Math.floor(Math.random()25)

if(!mines.includes(r)){
mines.push(r)
}

}

createGrid()

}

function openCell(i,cell){

if(!active)return

if(mines.includes(i)){

cell.innerText="💣"
cell.style.background="red"

alert("Вы проиграли")

active=false
return

}

cell.innerText="💎"
cell.style.background="green"

opened++

let win=bet(1+opened*0.4)

document.getElementById("profit").innerText=win.toFixed(2)

}

function cashout(){

if(!active)return

let win=Number(document.getElementById("profit").innerText)

balance+=win

updateBalance()

alert("Вы забрали $"+win)

active=false

}

createGrid()

updateBalance()

</script>
